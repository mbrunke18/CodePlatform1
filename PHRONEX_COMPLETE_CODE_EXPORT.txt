import express from 'express';
import session from 'express-session';
import ConnectPgSimple from 'connect-pg-simple';
import cors from 'cors';
import { createServer } from 'http';
import { WebSocketServer } from 'ws';
import pino from 'pino';
import pinoHttp from 'pino-http';
import passport from 'passport';
import { setupAuth } from './replitAuth.js';
import { db } from './db.js';
import { router } from './routes.js';
import { collaborationService } from './collaboration-service.js';
import { startBackgroundJobs } from './background-jobs.js';
import { auditLogger } from './middleware/audit-logging.js';
import { setupSwagger } from './swagger.js';
import { startProactiveAIRadar } from './proactive-ai-radar.js';

const logger = pino({
  level: 'info',
  transport: {
    target: 'pino-pretty',
    options: {
      colorize: true
    }
  }
});

const app = express();
const httpServer = createServer(app);
const PORT = process.env.PORT || 5000;

// PostgreSQL session store
const PgSession = ConnectPgSimple(session);

// Middleware
app.use(pinoHttp({ logger }));
app.use(cors({
  origin: process.env.NODE_ENV === 'production' ? false : ['http://localhost:5173', 'http://localhost:5000'],
  credentials: true
}));
app.use(express.json({ limit: '50mb' }));
app.use(express.urlencoded({ extended: true, limit: '50mb' }));

// Session configuration
app.use(session({
  store: new PgSession({
    pool: db,
    createTableIfMissing: true,
  }),
  secret: process.env.SESSION_SECRET || 'fallback-secret-key',
  resave: false,
  saveUninitialized: false,
  cookie: {
    secure: process.env.NODE_ENV === 'production',
    httpOnly: true,
    maxAge: 30 * 24 * 60 * 60 * 1000, // 30 days
  },
}));

// Passport middleware
app.use(passport.initialize());
app.use(passport.session());

// Setup authentication
setupAuth();

// Audit logging middleware
app.use(auditLogger);

// API routes
app.use('/api', router);

// Setup Swagger documentation
setupSwagger(app);

// WebSocket server for real-time collaboration
const wss = new WebSocketServer({ 
  server: httpServer, 
  path: '/ws' 
});

collaborationService.initialize(wss);

// Serve static files in production
if (process.env.NODE_ENV === 'production') {
  const path = await import('path');
  const { fileURLToPath } = await import('url');
  const __dirname = path.dirname(fileURLToPath(import.meta.url));
  
  app.use(express.static(path.join(__dirname, 'public')));
  
  app.get('*', (req, res) => {
    res.sendFile(path.join(__dirname, 'public/index.html'));
  });
}

// Start background jobs
startBackgroundJobs();

// Start proactive AI radar
startProactiveAIRadar();

// Error handling
app.use((err: any, req: express.Request, res: express.Response, next: express.NextFunction) => {
  logger.error(err);
  res.status(500).json({ 
    error: 'Internal server error',
    message: process.env.NODE_ENV === 'development' ? err.message : 'Something went wrong'
  });
});

// Start server
httpServer.listen(PORT, '0.0.0.0', () => {
  logger.info(`🚀 Phronex Kairosync Strategic Intelligence Platform running on port ${PORT}`);
  logger.info(`📊 Environment: ${process.env.NODE_ENV || 'development'}`);
  logger.info(`🔒 Database: Connected to PostgreSQL`);
  logger.info(`🌐 WebSocket: Real-time collaboration enabled`);
  logger.info(`🤖 AI Intelligence: All modules active`);
  logger.info(`📈 Background Jobs: Processing system intelligence`);
});

export default app;
