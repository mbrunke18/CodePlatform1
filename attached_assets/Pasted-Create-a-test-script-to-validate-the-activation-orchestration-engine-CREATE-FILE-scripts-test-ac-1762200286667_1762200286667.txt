import { db } from '../db/client';
import { eq } from 'drizzle-orm';
import { playbookLibrary, scenarioStakeholders, executionInstances, notifications } from '../db/schema';

async function testActivation() {
  console.log('üß™ Testing Activation Engine...\n');
  
  // Step 1: Find a playbook to test with
  const playbook = await db.query.playbookLibrary.findFirst({
    where: eq(playbookLibrary.domain, 'Crisis Management'),
    limit: 1
  });
  
  if (!playbook) {
    console.error('‚ùå No playbook found');
    return;
  }
  
  console.log(`‚úÖ Found playbook: ${playbook.name}`);
  
  // Step 2: Test orchestration endpoint
  console.log('\nüì° Calling /api/activations/orchestrate...');
  
  const response = await fetch('http://localhost:3000/api/activations/orchestrate', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      playbookId: playbook.id,
      triggerId: 'test-trigger-123',
      executedBy: '7cd941d8-5c5f-461e-87ea-9d2b1d81cb59', // Demo user
      context: {
        testMode: true,
        scenario: 'End-to-end validation'
      }
    })
  });
  
  if (!response.ok) {
    console.error('‚ùå Orchestration failed:', await response.text());
    return;
  }
  
  const result = await response.json();
  console.log('‚úÖ Orchestration response:', JSON.stringify(result, null, 2));
  
  const executionInstanceId = result.executionInstanceId;
  
  // Step 3: Verify execution instance created
  console.log('\nüîç Checking database...');
  
  const execution = await db.query.executionInstances.findFirst({
    where: eq(executionInstances.id, executionInstanceId)
  });
  
  if (!execution) {
    console.error('‚ùå Execution instance not found in database');
    return;
  }
  
  console.log(`‚úÖ Execution instance created:`, {
    id: execution.id,
    status: execution.status,
    startedAt: execution.startedAt
  });
  
  // Step 4: Verify notifications created
  const notifs = await db.query.notifications.findMany({
    where: eq(notifications.metadata.executionInstanceId, executionInstanceId)
  });
  
  console.log(`‚úÖ Notifications created: ${notifs.length}`);
  console.log('   Sample notification:', {
    title: notifs[0]?.title,
    status: notifs[0]?.status,
    recipientId: notifs[0]?.recipientId
  });
  
  // Step 5: Test status endpoint
  console.log('\nüì° Calling /api/activations/status...');
  
  const statusResponse = await fetch(`http://localhost:3000/api/activations/${executionInstanceId}/status`);
  
  if (!statusResponse.ok) {
    console.error('‚ùå Status endpoint failed');
    return;
  }
  
  const status = await statusResponse.json();
  console.log('‚úÖ Status response:', JSON.stringify(status, null, 2));
  
  // Step 6: Test acknowledgment
  console.log('\nüì° Testing acknowledgment...');
  
  if (notifs.length > 0) {
    const ackResponse = await fetch(`http://localhost:3000/api/notifications/${notifs[0].id}/acknowledge`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' }
    });
    
    if (ackResponse.ok) {
      const ackResult = await ackResponse.json();
      console.log('‚úÖ Acknowledgment successful:', ackResult);
    } else {
      console.error('‚ùå Acknowledgment failed');
    }
  }
  
  console.log('\n‚ú® Test complete!\n');
  console.log('SUMMARY:');
  console.log(`- Playbook: ${playbook.name}`);
  console.log(`- Execution Instance: ${executionInstanceId}`);
  console.log(`- Notifications Created: ${notifs.length}`);
  console.log(`- Status: ${status.coordination?.coordinationComplete ? 'COORDINATED' : 'IN PROGRESS'}`);
}

testActivation().catch(console.error);
