import { Card, CardContent, CardHeader, CardTitle, CardDescription } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Badge } from "@/components/ui/badge";
import { Input } from "@/components/ui/input";
import StandardNav from "@/components/layout/StandardNav";
import { useQuery } from "@tanstack/react-query";
import { useLocation } from "wouter";
import { Search, Plus, Eye, Copy, Edit } from "lucide-react";
import type { Playbook } from "@shared/schema";

export default function PlaybookLibrary() {
  const [, setLocation] = useLocation();
  
  // Fetch templates (isTemplate = true)
  const { data: templates } = useQuery<Playbook[]>({
    queryKey: ['/api/playbooks/templates'],
  });
  
  // Fetch user's custom playbooks
  const { data: myPlaybooks } = useQuery<Playbook[]>({
    queryKey: ['/api/playbooks'],
  });
  
  const categoryColors = {
    offense: 'bg-emerald-100 text-emerald-700 dark:bg-emerald-900/30 dark:text-emerald-300',
    defense: 'bg-blue-100 text-blue-700 dark:bg-blue-900/30 dark:text-blue-300',
    special_teams: 'bg-purple-100 text-purple-700 dark:bg-purple-900/30 dark:text-purple-300'
  };
  
  return (
    <div className="min-h-screen bg-slate-50 dark:bg-slate-900">
      <StandardNav />
      
      <div className="container mx-auto px-4 py-20">
        {/* Header */}
        <div className="flex items-center justify-between mb-8">
          <div>
            <h1 className="text-4xl font-bold text-slate-900 dark:text-white mb-2">
              Playbook Library
            </h1>
            <p className="text-lg text-slate-600 dark:text-slate-400">
              166 battle-tested templates. Use as-is, customize, or build your own.
            </p>
          </div>
          <Button size="lg" onClick={() => setLocation('/playbooks/create')}>
            <Plus className="mr-2 h-5 w-5" />
            Create Custom Playbook
          </Button>
        </div>
        
        {/* Search & Filter */}
        <div className="flex gap-4 mb-8">
          <div className="flex-1 relative">
            <Search className="absolute left-3 top-1/2 -translate-y-1/2 h-5 w-5 text-slate-400" />
            <Input 
              placeholder="Search playbooks..." 
              className="pl-10"
              data-testid="input-search"
            />
          </div>
          <Button variant="outline">Filter by Domain</Button>
          <Button variant="outline">Filter by Category</Button>
        </div>
        
        {/* My Custom Playbooks */}
        {myPlaybooks && myPlaybooks.length > 0 && (
          <div className="mb-12">
            <h2 className="text-2xl font-bold text-slate-900 dark:text-white mb-4">
              My Custom Playbooks ({myPlaybooks.length})
            </h2>
            <div className="grid md:grid-cols-2 lg:grid-cols-3 gap-6">
              {myPlaybooks.map((playbook) => (
                <PlaybookCard 
                  key={playbook.id} 
                  playbook={playbook}
                  isCustom={true}
                  categoryColors={categoryColors}
                />
              ))}
            </div>
          </div>
        )}
        
        {/* Pre-Built Templates */}
        <div>
          <h2 className="text-2xl font-bold text-slate-900 dark:text-white mb-4">
            Pre-Built Templates (166)
          </h2>
          
          {/* Group by category */}
          {['offense', 'defense', 'special_teams'].map((category) => {
            const categoryPlaybooks = templates?.filter(p => p.category === category) || [];
            const categoryLabel = category === 'offense' ? 'OFFENSE - Seize Opportunities' :
                                 category === 'defense' ? 'DEFENSE - Protect Value' :
                                 'SPECIAL TEAMS - Change the Game';
            
            return (
              <div key={category} className="mb-8">
                <h3 className="text-xl font-bold text-slate-700 dark:text-slate-300 mb-4">
                  {categoryLabel} ({categoryPlaybooks.length})
                </h3>
                <div className="grid md:grid-cols-2 lg:grid-cols-3 gap-6">
                  {categoryPlaybooks.map((playbook) => (
                    <PlaybookCard 
                      key={playbook.id} 
                      playbook={playbook}
                      isCustom={false}
                      categoryColors={categoryColors}
                    />
                  ))}
                </div>
              </div>
            );
          })}
        </div>
      </div>
    </div>
  );
}

function PlaybookCard({ 
  playbook, 
  isCustom,
  categoryColors 
}: { 
  playbook: Playbook; 
  isCustom: boolean;
  categoryColors: Record<string, string>;
}) {
  const [, setLocation] = useLocation();
  
  return (
    <Card className="hover:shadow-lg transition-shadow" data-testid={`card-playbook-${playbook.id}`}>
      <CardHeader>
        <div className="flex items-start justify-between mb-2">
          <Badge className={categoryColors[playbook.category || 'offense']}>
            {playbook.category?.replace('_', ' ').toUpperCase()}
          </Badge>
          {isCustom && <Badge variant="outline">Custom</Badge>}
        </div>
        <CardTitle className="text-lg">{playbook.name}</CardTitle>
        <CardDescription>{playbook.description}</CardDescription>
      </CardHeader>
      <CardContent>
        <div className="space-y-2 text-sm text-slate-600 dark:text-slate-400 mb-4">
          {!isCustom && (
            <>
              <div>‚≠ê Used by {playbook.timesUsed || 0} organizations</div>
              <div>‚è±Ô∏è Avg Response: {playbook.avgResponseTimeSeconds ? `${Math.round(playbook.avgResponseTimeSeconds / 60)} min` : 'N/A'}</div>
            </>
          )}
          <div>üìÅ Domain: {playbook.domain}</div>
        </div>
        
        <div className="flex gap-2">
          <Button 
            variant="outline" 
            size="sm" 
            className="flex-1"
            onClick={() => setLocation(`/playbooks/${playbook.id}/preview`)}
          >
            <Eye className="mr-1 h-4 w-4" />
            Preview
          </Button>
          
          {isCustom ? (
            <Button 
              size="sm" 
              className="flex-1"
              onClick={() => setLocation(`/playbooks/${playbook.id}/edit`)}
            >
              <Edit className="mr-1 h-4 w-4" />
              Edit
            </Button>
          ) : (
            <>
              <Button 
                variant="outline"
                size="sm"
                onClick={() => setLocation(`/playbooks/${playbook.id}/customize`)}
              >
                <Edit className="mr-1 h-4 w-4" />
                Customize
              </Button>
              <Button 
                size="sm"
                onClick={() => setLocation(`/playbooks/${playbook.id}/clone`)}
              >
                <Copy className="mr-1 h-4 w-4" />
                Clone
              </Button>
            </>
          )}
        </div>
      </CardContent>
    </Card>
  );
}
