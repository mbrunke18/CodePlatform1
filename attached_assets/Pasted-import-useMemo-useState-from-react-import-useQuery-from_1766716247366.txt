import { useMemo, useState } from "react";
import { useQuery } from "@tanstack/react-query";
import { useLocation } from "wouter";

import StandardNav from "@/components/layout/StandardNav";
import Footer from "@/components/layout/Footer";

import { Card, CardContent, CardHeader, CardTitle, CardDescription } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Badge } from "@/components/ui/badge";
import { Input } from "@/components/ui/input";

import {
  ArrowRight,
  Layers,
  BookOpen,
  Shield,
  Zap,
  Radar,
  Radio,
  BarChart3,
} from "lucide-react";

import type { Playbook } from "@shared/schema";

/**
 * FINAL Canonical Playbook Library
 * - Search
 * - Stats
 * - Category grouping
 * - IDEA phase coloring
 * - Empty & loading states
 * - data-testid everywhere
 */

const CATEGORIES = {
  offense: { label: "OFFENSE", tagline: "Seize Opportunities", icon: Zap },
  defense: { label: "DEFENSE", tagline: "Protect Value", icon: Shield },
  special_teams: { label: "SPECIAL TEAMS", tagline: "Change the Game", icon: BookOpen },
} as const;

type Category = keyof typeof CATEGORIES;

const PHASES = {
  identify: {
    label: "Identify",
    icon: BookOpen,
    badge: "bg-violet-100 text-violet-700 dark:bg-violet-900/40 dark:text-violet-300",
  },
  detect: {
    label: "Detect",
    icon: Radar,
    badge: "bg-blue-100 text-blue-700 dark:bg-blue-900/40 dark:text-blue-300",
  },
  execute: {
    label: "Execute",
    icon: Radio,
    badge: "bg-emerald-100 text-emerald-700 dark:bg-emerald-900/40 dark:text-emerald-300",
  },
  advance: {
    label: "Advance",
    icon: BarChart3,
    badge: "bg-amber-100 text-amber-700 dark:bg-amber-900/40 dark:text-amber-300",
  },
} as const;

type PhaseKey = keyof typeof PHASES;

export default function PlaybookLibrary() {
  const [, setLocation] = useLocation();
  const [search, setSearch] = useState("");

  const { data, isLoading } = useQuery<Playbook[]>({
    queryKey: ["/api/playbooks"],
  });

  const filtered = useMemo(() => {
    if (!data) return [];
    const q = search.toLowerCase();
    return data.filter(
      (pb) =>
        pb.name.toLowerCase().includes(q) ||
        pb.domain.toLowerCase().includes(q)
    );
  }, [data, search]);

  const grouped = useMemo(() => {
    const base: Record<Category, Playbook[]> = {
      offense: [],
      defense: [],
      special_teams: [],
    };

    filtered.forEach((pb) => {
      if (!pb.category) return;
      if (pb.category in base) {
        base[pb.category as Category].push(pb);
      }
    });

    return base;
  }, [filtered]);

  const totalCount = filtered.length;

  return (
    <>
      <StandardNav />

      <main className="max-w-7xl mx-auto px-6 py-12">
        {/* Header */}
        <header className="mb-10" data-testid="playbook-library-header">
          <h1 className="text-4xl font-bold text-slate-900 dark:text-white">
            Playbook Library
          </h1>
          <p className="text-slate-600 dark:text-slate-400 mt-2">
            Execute strategy at the speed of disruption.
          </p>
        </header>

        {/* Search */}
        <div className="mb-10 max-w-md" data-testid="search-container">
          <Input
            placeholder="Search playbooks or domains…"
            value={search}
            onChange={(e) => setSearch(e.target.value)}
            data-testid="input-search"
          />
        </div>

        {/* Stats */}
        {!isLoading && (
          <div
            className="grid grid-cols-2 md:grid-cols-4 gap-6 mb-14"
            data-testid="stats-bar"
          >
            <Stat label="Total Playbooks" value={totalCount} />
            <Stat label="Offense" value={grouped.offense.length} />
            <Stat label="Defense" value={grouped.defense.length} />
            <Stat label="Special Teams" value={grouped.special_teams.length} />
          </div>
        )}

        {/* Loading */}
        {isLoading && (
          <div
            className="py-24 text-center text-slate-500"
            data-testid="loading-state"
          >
            <Layers className="h-8 w-8 mx-auto mb-3 animate-pulse" />
            Loading playbooks…
          </div>
        )}

        {/* Empty */}
        {!isLoading && totalCount === 0 && (
          <div
            className="py-24 text-center text-slate-500"
            data-testid="empty-state"
          >
            No playbooks match your search.
          </div>
        )}

        {/* Categories */}
        {!isLoading &&
          (Object.keys(CATEGORIES) as Category[]).map((category) => {
            const meta = CATEGORIES[category];
            const playbooks = grouped[category];

            if (playbooks.length === 0) return null;

            return (
              <section
                key={category}
                className="mb-16"
                data-testid={`section-${category}`}
              >
                <div className="flex items-center justify-between mb-6">
                  <div className="flex items-center gap-3">
                    <meta.icon className="h-6 w-6 text-slate-500" />
                    <div>
                      <h2 className="text-2xl font-bold text-slate-900 dark:text-white">
                        {meta.label}
                      </h2>
                      <p className="text-slate-500">{meta.tagline}</p>
                    </div>
                  </div>

                  <Badge variant="secondary" data-testid={`count-${category}`}>
                    {playbooks.length}
                  </Badge>
                </div>

                <div className="grid gap-6 sm:grid-cols-2 lg:grid-cols-3">
                  {playbooks.map((playbook) => {
                    const phaseKey =
                      (playbook as any).phase as PhaseKey | undefined;
                    const phase = phaseKey ? PHASES[phaseKey] : null;

                    return (
                      <Card
                        key={playbook.id}
                        className="cursor-pointer transition hover:shadow-lg"
                        onClick={() =>
                          setLocation(`/playbooks/${playbook.id}`)
                        }
                        data-testid={`card-playbook-${playbook.id}`}
                      >
                        <CardHeader className="space-y-2">
                          <CardTitle>{playbook.name}</CardTitle>
                          <CardDescription>{playbook.domain}</CardDescription>

                          {phase && (
                            <Badge
                              className={`w-fit ${phase.badge}`}
                              data-testid={`badge-phase-${playbook.id}`}
                            >
                              {phase.label}
                            </Badge>
                          )}
                        </CardHeader>

                        <CardContent className="space-y-4">
                          {playbook.description && (
                            <p className="text-sm text-slate-600 dark:text-slate-400">
                              {playbook.description}
                            </p>
                          )}

                          <Button
                            variant="ghost"
                            className="w-full justify-between"
                            data-testid={`button-open-${playbook.id}`}
                          >
                            Open Playbook
                            <ArrowRight className="h-4 w-4" />
                          </Button>
                        </CardContent>
                      </Card>
                    );
                  })}
                </div>
              </section>
            );
          })}
      </main>

      <Footer />
    </>
  );
}

function Stat({ label, value }: { label: string; value: number }) {
  return (
    <div className="text-center" data-testid={`stat-${label}`}>
      <div className="text-2xl font-bold text-slate-900 dark:text-white">
        {value}
      </div>
      <div className="text-sm text-slate-500">{label}</div>
    </div>
  );
}
