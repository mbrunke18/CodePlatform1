Everything You Need to Build Decision + Execution System

WHAT WE'RE BUILDING: THE COMPLETE PICTURE
TRIGGER → DECISION → EXECUTION → VERIFICATION
(0-60s)   (1-15min)   (15-90min)    (90-120min)

Example: Data Breach Detected
├─ TRIGGER: Breach alert fires → System identifies scenario
├─ DECISION: CEO chooses "Disclose immediately" (14 min)
├─ EXECUTION: 47 tasks coordinated across 12 stakeholders (60 min)
└─ VERIFICATION: All tasks complete, outcome logged, lessons captured

typescriptimport { pgTable, varchar, text, uuid, timestamp, jsonb, integer, boolean } from "drizzle-orm/pg-core";
import { createInsertSchema } from "drizzle-zod";
import { z } from "zod";

export const users = pgTable('users', {
  id: uuid('id').primaryKey().defaultRandom(),
  username: varchar('username', { length: 255 }).notNull().unique(),
  password: varchar('password', { length: 255 }).notNull(),
  createdAt: timestamp('created_at').defaultNow(),
});

export const organizations = pgTable('organizations', {
  id: uuid('id').primaryKey().defaultRandom(),
  name: varchar('name', { length: 255 }).notNull(),
  pilotTier: varchar('pilot_tier', { length: 50 }),
  pilotStartDate: timestamp('pilot_start_date'),
  pilotEndDate: timestamp('pilot_end_date'),
  pilotValue: integer('pilot_value'),
  industry: varchar('industry', { length: 100 }),
  isBeachheadCustomer: boolean('is_beachhead_customer').default(false),
  avgResponseTimeSeconds: integer('avg_response_time_seconds'),
  totalExecutions: integer('total_executions').default(0),
  gameDayReadinessScore: integer('game_day_readiness_score'),
  dedicatedCsmId: uuid('dedicated_csm_id'),
  avgDecisionTimeMinutes: integer('avg_decision_time_minutes'),
  totalDecisionsMade: integer('total_decisions_made').default(0),
  createdAt: timestamp('created_at').defaultNow(),
  updatedAt: timestamp('updated_at').defaultNow(),
});

export const playbooks = pgTable('playbooks', {
  id: uuid('id').primaryKey().defaultRandom(),
  organizationId: uuid('organization_id').notNull(),
  name: varchar('name', { length: 255 }).notNull(),
  domain: varchar('domain', { length: 100 }).notNull(),
  category: varchar('category', { length: 50 }),
  description: text('description'),
  sourceType: varchar('source_type', { length: 50 }).notNull().default('custom'),
  templateId: uuid('template_id'),
  timesUsed: integer('times_used').default(0),
  avgExecutionTimeMinutes: integer('avg_execution_time_minutes'),
  
  // Task structure for execution
  tasks: jsonb('tasks').$type<Array<{
    id: string;
    title: string;
    description: string;
    assignee: string; // email or role
    timeTargetMinutes: number;
    order: number;
    isParallel: boolean; // Can execute simultaneously with others
    dependencies: string[]; // Task IDs that must complete first
    template?: string; // Pre-written template for this task
    validationRequired: boolean; // Needs approval before proceeding
  }>>(),
  
  isActive: boolean('is_active').default(true),
  isTemplate: boolean('is_template').default(false),
  createdAt: timestamp('created_at').defaultNow(),
  updatedAt: timestamp('updated_at').defaultNow(),
});

export const decisionTrees = pgTable('decision_trees', {
  id: uuid('id').primaryKey().defaultRandom(),
  organizationId: uuid('organization_id').notNull(),
  name: varchar('name', { length: 255 }).notNull(),
  scenario: varchar('scenario', { length: 255 }).notNull(),
  domain: varchar('domain', { length: 100 }).notNull(),
  category: varchar('category', { length: 50 }),
  
  decisionPoints: jsonb('decision_points').$type<Array<{
    id: string;
    order: number;
    question: string;
    decisionMaker: string;
    mustWeighIn: string[];
    timeWindowMinutes: number;
    options: Array<{
      id: string;
      label: string;
      description: string;
      pros: string[];
      cons: string[];
      criteria: Array<{condition: string; met: boolean | null;}>;
      triggersPlaybookId?: string; // Links to playbook for execution
    }>;
  }>>(),
  
  isActive: boolean('is_active').default(true),
  timesUsed: integer('times_used').default(0),
  avgDecisionTimeMinutes: integer('avg_decision_time_minutes'),
  createdAt: timestamp('created_at').defaultNow(),
  updatedAt: timestamp('updated_at').defaultNow(),
});

// Active coordinated response (decision + execution)
export const activeResponses = pgTable('active_responses', {
  id: uuid('id').primaryKey().defaultRandom(),
  organizationId: uuid('organization_id').notNull(),
  decisionTreeId: uuid('decision_tree_id').references(() => decisionTrees.id),
  
  // PHASE 1: TRIGGER
  scenarioName: varchar('scenario_name', { length: 255 }).notNull(),
  triggeredAt: timestamp('triggered_at').defaultNow(),
  triggeredBy: varchar('triggered_by', { length: 255 }),
  
  // PHASE 2: DECISION
  currentPhase: varchar('current_phase', { length: 50 }).default('decision'), // decision | execution | complete
  decisionMaker: varchar('decision_maker', { length: 255 }),
  decisionQuestion: text('decision_question'),
  decisionMadeAt: timestamp('decision_made_at'),
  decisionTimeMinutes: integer('decision_time_minutes'),
  optionChosen: varchar('option_chosen', { length: 255 }),
  
  stakeholderInputs: jsonb('stakeholder_inputs').$type<Array<{
    userId: string;
    role: string;
    input: string;
    timestamp: string;
    recommendedOption: string;
  }>>(),
  
  // PHASE 3: EXECUTION
  playbookId: uuid('playbook_id').references(() => playbooks.id),
  executionStartedAt: timestamp('execution_started_at'),
  executionCompletedAt: timestamp('execution_completed_at'),
  executionTimeMinutes: integer('execution_time_minutes'),
  
  taskStatuses: jsonb('task_statuses').$type<Array<{
    taskId: string;
    status: 'pending' | 'in_progress' | 'completed' | 'blocked';
    assignee: string;
    startedAt?: string;
    completedAt?: string;
    timeSpentMinutes?: number;
    output?: string; // What was produced (e.g., draft statement)
  }>>(),
  
  // PHASE 4: VERIFICATION
  status: varchar('status', { length: 50 }).default('active'), // active | completed | failed
  totalResponseTimeMinutes: integer('total_response_time_minutes'),
  outcome: varchar('outcome', { length: 100 }), // positive | neutral | negative
  lessons: text('lessons'),
  completedAt: timestamp('completed_at'),
});

export const decisionLog = pgTable('decision_log', {
  id: uuid('id').primaryKey().defaultRandom(),
  organizationId: uuid('organization_id').notNull(),
  activeResponseId: uuid('active_response_id').references(() => activeResponses.id),
  scenario: varchar('scenario', { length: 255 }).notNull(),
  question: text('question').notNull(),
  decisionMaker: varchar('decision_maker', { length: 255 }).notNull(),
  optionChosen: text('option_chosen').notNull(),
  decisionTimeMinutes: integer('decision_time_minutes').notNull(),
  executionTimeMinutes: integer('execution_time_minutes'),
  totalResponseTimeMinutes: integer('total_response_time_minutes'),
  outcome: varchar('outcome', { length: 100 }),
  lessons: text('lessons'),
  timestamp: timestamp('timestamp').defaultNow(),
});

export const milestoneTracking = pgTable('milestone_tracking', {
  id: uuid('id').primaryKey().defaultRandom(),
  quarter: varchar('quarter', { length: 10 }).notNull(),
  metric: varchar('metric', { length: 100 }).notNull(),
  goal: integer('goal').notNull(),
  current: integer('current').default(0),
  unit: varchar('unit', { length: 10 }),
  updatedAt: timestamp('updated_at').defaultNow(),
  notes: text('notes')
});

export const insertUserSchema = createInsertSchema(users).omit({id: true, createdAt: true});

// Update task status
app.patch('/api/active-responses/:id/tasks/:taskId', optionalAuth, async (req, res) => {
  try {
    const { status, output, completedAt } = req.body;
    
    // Fetch current response
    const [response] = await db.select()
      .from(activeResponses)
      .where(eq(activeResponses.id, req.params.id));
    
    if (!response) {
      return res.status(404).json({ error: 'Response not found' });
    }
    
    // Update task status in JSONB
    const taskStatuses = response.taskStatuses as any[] || [];
    const taskIndex = taskStatuses.findIndex(t => t.taskId === req.params.taskId);
    
    if (taskIndex === -1) {
      return res.status(404).json({ error: 'Task not found' });
    }
    
    taskStatuses[taskIndex] = {
      ...taskStatuses[taskIndex],
      status,
      output,
      completedAt,
      timeSpentMinutes: completedAt 
        ? Math.round((new Date(completedAt).getTime() - new Date(taskStatuses[taskIndex].startedAt).getTime()) / 60000)
        : undefined
    };
    
    // Update database
    const [updated] = await db.update(activeResponses)
      .set({ taskStatuses })
      .where(eq(activeResponses.id, req.params.id))
      .returning();
    
    res.json(updated);
  } catch (error) {
    res.status(500).json({ error: 'Failed to update task' });
  }
});
